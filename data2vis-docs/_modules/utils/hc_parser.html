[docs]def convert_vl_to_hc(decoded_post, source_data):
    """
    FUNC: Convert Vega-lite JSON specs to Highcharts specs

    Args:
        decoded_post (string): The JSONified Vega-lite specification
        source_data (list): List of records in dataset.
            The records are in the form of dictionaries where the column names
            are keys and the corresponding data points are values.
    
    Returns:
        string: hc_specs - The JSONified Highcharts specification
    """
    try:
        specification = json.loads(decoded_post)
        print()
        print("Vega-lite: ", specification)
        print(type(specification))
    except ValueError:
        specification = None
        print()
        print("Error reading Vega-lite JSON specifications")
        print("JSON Specs: ", decoded_post)
        return None

    source_data_df = pd.DataFrame(source_data)

    # Decode Vega-lite specs
    try:
        decoded_specs, source_data_df = decode_vega_specs(specification, source_data_df)
    except:
        print("Error decoding Vega-lite JSON specifications")
        return None

    # Aggregate the sourcedata based on the decoded Vega-lite specs
    try:
        decoded_specs, intermediary_data = get_intermediary_data(decoded_specs, source_data_df)
    except:
        print("Error aggregating the source data")
        return None

    # Generate the Highcharts specs based on decode Vega-lite specs and aggregated data
    try:
        hc_specs = create_hc_specs(decoded_specs, intermediary_data)
    except:
        print("Error generating the Vega-lite JSON specifications")
        return None

    return hc_specs

[docs]def get_intermediary_data(decoded_specs, source_data_df):
    """
    FUNC: Getting the intermediary data from the source data. The intermediary data is computed based on
        all the aggregation and binning of the relevant fields mentioned in the generated specification

    Args:
        decoded_specs (tuple): Extracted components of the Vega-lite specifications
        source_data_df (DataFrame): Processed input dataset
    
    Returns:
        tuple:\n
            tuple: decoded_specs - Extracted components of the Vega-lite specifications\n
            DataFrame: intermediary_data - Intermediary data that needs to be plotted
    """
    chart_type, x_field_name, x_aggregation, y_field_name, y_aggregation, detail_field_name, color_field_name, size_field_name, shape_field_name = decoded_specs

    if detail_field_name is not None:
        groupby_field_name = detail_field_name
    elif color_field_name:
        groupby_field_name = color_field_name
    elif size_field_name:
        groupby_field_name = size_field_name
    elif shape_field_name:
        groupby_field_name = shape_field_name
    else:
        groupby_field_name = None

    # Calculating the intermediary data
    intermediary_data = None
    if x_field_name is not None:
        if y_field_name is not None:
            if groupby_field_name is not None:
                if y_aggregation in ['count', 'sum', 'mean']:
                    agg_func = {
                                   y_field_name: {
                                      y_aggregation + "(" + y_field_name + ")": y_aggregation
                                   }
                               }
                    # Mean & sum might throw exceptions
                    if y_aggregation in ['sum', 'mean']:
                        source_data_df[y_field_name] = source_data_df[y_field_name].astype('float64')
                    intermediary_data = source_data_df.groupby([x_field_name, groupby_field_name]).agg(agg_func)
                    intermediary_data.columns = intermediary_data.columns.droplevel()
                    intermediary_data = intermediary_data.sort_index().reset_index()
                    y_field_name = y_aggregation + "(" + y_field_name + ")"
                else:
                    print("Faulty Vega config")
            else:
                if y_aggregation in ['count', 'sum', 'mean']:
                    agg_func = {
                                   y_field_name: {
                                      y_aggregation + "(" + y_field_name + ")": y_aggregation
                                   }
                               }
                    # Mean & sum might throw exceptions
                    if y_aggregation in ['sum', 'mean']:
                        source_data_df[y_field_name] = source_data_df[y_field_name].astype('float64')
                    intermediary_data = source_data_df.groupby([x_field_name]).agg(agg_func)
                    intermediary_data.columns = intermediary_data.columns.droplevel()
                    intermediary_data = intermediary_data.sort_index().reset_index()
                    y_field_name = y_aggregation + "(" + y_field_name + ")"
                else:
                    print("Faulty Vega config")
        else:
            print("Faulty Vega config")
    else:
        print()
        print("Faulty Vega config")

    print()
    print("HC chart data: ", intermediary_data)

    decoded_specs = chart_type, x_field_name, x_aggregation, y_field_name, y_aggregation, detail_field_name, color_field_name, size_field_name, shape_field_name

    return decoded_specs, intermediary_data



[docs]def create_hc_specs(decoded_specs, intermediary_data):
    """
    FUNC: Generating the Highcharts specifications from the decoded specs and intermediary dataset

    Args:
        decoded_specs (tuple): Extracted components of the Vega-lite specifications
        intermediary_data (DataFrame): Intermediary data that actually needs to be plotted

    Returns:
        string: temp - JSONified Highcharts specifications
    """
    chart_type, x_field_name, x_aggregation, y_field_name, y_aggregation, detail_field_name, color_field_name, size_field_name, shape_field_name = decoded_specs

    # Createing the highcharts specifications
    temp = None
    if intermediary_data is not None:
        # Chart Type
        temp = dict()
        temp["chart"] = dict()
        temp["chart"]["type"] = chart_type

        # xAxis
        temp["xAxis"] = dict()
        temp["xAxis"]["title"] = dict()
        temp["xAxis"]["title"]["text"] = x_field_name

        temp["xAxis"]["categories"] = list(intermediary_data[x_field_name].astype(str).unique())

        # yAxis
        temp["yAxis"] = dict()
        temp["yAxis"]["title"] = dict()
        temp["yAxis"]["title"]["text"] = y_field_name

        # Series data
        temp["series"] = []
        if detail_field_name is not None:
            intermediary_data = intermediary_data.pivot(index=x_field_name, columns=detail_field_name, values=y_field_name)
            intermediary_data = intermediary_data.fillna(0)
            intermediary_data.columns = intermediary_data.columns.astype(str).values.tolist()
            for col in intermediary_data.columns.values.tolist():
                data = dict()
                data["name"] = col
                data["data"] = list(intermediary_data[col])
                temp["series"].append(data)

            # Stack the different series for area and column graphs
            if temp["chart"]["type"] in ["area", "column"]:
                temp["plotOptions"] = dict()
                temp["plotOptions"]["series"] = dict()
                temp["plotOptions"]["series"]["stacking"] = "normal"
        elif color_field_name is not None:
            intermediary_data = intermediary_data.pivot(index=x_field_name, columns=color_field_name, values=y_field_name)
            intermediary_data = intermediary_data.fillna(0)
            intermediary_data.columns = intermediary_data.columns.astype(str).values.tolist()
            for col in intermediary_data.columns.values.tolist():
                data = dict()
                data["name"] = col
                data["data"] = list(intermediary_data[col])
                temp["series"].append(data)

            # Stack the different series for area and column graphs
            if temp["chart"]["type"] in ["area", "column"]:
                temp["plotOptions"] = dict()
                temp["plotOptions"]["series"] = dict()
                temp["plotOptions"]["series"]["stacking"] = "normal"
        elif size_field_name is not None:
            intermediary_data = intermediary_data.pivot(index=x_field_name, columns=size_field_name, values=y_field_name)
            intermediary_data = intermediary_data.fillna(0)
            intermediary_data.columns = intermediary_data.columns.astype(str).values.tolist()
            for col in intermediary_data.columns.values.tolist():
                data = dict()
                data["name"] = col
                data["data"] = list(intermediary_data[col])
                temp["series"].append(data)

            # Stack the different series for area and column graphs
            if temp["chart"]["type"] in ["area", "column"]:
                temp["plotOptions"] = dict()
                temp["plotOptions"]["series"] = dict()
                temp["plotOptions"]["series"]["stacking"] = "normal"
        elif shape_field_name is not None:
            intermediary_data = intermediary_data.pivot(index=x_field_name, columns=shape_field_name, values=y_field_name)
            intermediary_data = intermediary_data.fillna(0)
            intermediary_data.columns = intermediary_data.columns.astype(str).values.tolist()
            for col in intermediary_data.columns.values.tolist():
                data = dict()
                data["name"] = col
                data["data"] = list(intermediary_data[col])
                temp["series"].append(data)

            # Stack the different series for area and column graphs
            if temp["chart"]["type"] in ["area", "column"]:
                temp["plotOptions"] = dict()
                temp["plotOptions"]["series"] = dict()
                temp["plotOptions"]["series"]["stacking"] = "normal"
        else:
            data = dict()
            data["name"] = y_field_name
            data["data"] = list(intermediary_data[y_field_name])
            temp["series"].append(data)

    temp = json.dumps(temp)

    print()
    print("HC specifications: ", temp)
    return temp

    """
    FUNC: Bin the field into appropriate number of buckets

    Args:

        field_name (string): Field that needs to be binned

    bin_count = 8
