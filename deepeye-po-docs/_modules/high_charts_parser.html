[docs]def bar_simple_category_get_values(data, x_axis_name, y_axis_name):
    """
    FUNC:
        - Given Chart type -> Bar, Sub-type -> Simple
        - find get x and y axes data   
    ARGS:
        data (dict): Jsonified data  
        x_axis_name (str): Unique name representing x-axis
        y_axis_name (str): Unique name representing y-axis 
    RETURNS:
        tuple: Returns x and y axes data
            - x_values
            - y_values
    """
    values = data["data"]["values"];
    x_values = [];
    y_values = [];
    for i in tqdm(range(len(values))):
        x_values.append(values[i][x_axis_name]);
        y_values.append(values[i][y_axis_name]);

    return x_values, y_values;

[docs]def scatter_multi_category_get_values(sub_data):
    """
    FUNC:
        - Given Chart type -> Scatter, Sub-type -> Multiple
        - find get all axes data   
    ARGS:
        sub_data (dict): Jsonified data from pre vega-lite encoded data 
    RETURNS:
        tuple: Returns 
            - x-y axes data in terms of corresponding pairs
            - 3rd axes data. 
    """
    x_values = sub_data['x_data'];
    y_values = sub_data['y_data'];
    x_y_value_pair = [];
    color_values = sub_data['classify'];
    for i in tqdm(range(len(x_values))):
        tmp = [];
        for j in tqdm(range(len(x_values[i]))):
            tmp.append([x_values[i][j], y_values[i][j]])
        x_y_value_pair.append(tmp);
    return x_y_value_pair, color_values;
